Integration Examples

Basic Picture Element

<picture>
  <source type="image/avif" srcset="avif/asset-name.avif">
  <source type="image/webp" 
          srcset="webp-mobile/asset-name_400w.webp 400w,
                  webp-medium/asset-name_800w.webp 800w,
                  webp/asset-name.webp 1600w"
          sizes="(max-width: 640px) 100vw, 50vw">
  <img src="webp/asset-name.webp" alt="Description" loading="lazy">
</picture>

With LQIP Placeholder

<div class="aether-image">
  <div class="lqip" style="background-image: url('lqip/asset-name-blur.webp')"></div>
  <img src="webp/asset-name.webp" 
       onload="this.classList.add('loaded'); this.parentElement.classList.add('loaded')">
</div>

Service Worker Registration

if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('/assets/aether-sw.js')
    .then(reg => console.log('AETHER SW registered'))
    .catch(err => console.error('SW registration failed:', err));
}
