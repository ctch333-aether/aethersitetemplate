‚öôÔ∏è CSS Integration

CSS Variables for Asset Paths

:root {
  --asset-base: '/assets';
  --asset-webp: '/assets/webp';
  --asset-png: '/assets/png-fallback';
  --asset-thumb: '/assets/thumbnails';
  --asset-medium: '/assets/webp-medium';
}

/* Background image with fallback */
.hero-section {
  background-image: url(var(--asset-png)/aether-portfolio-enchant-lights-01.png);
  background-image: image-set(
    url(var(--asset-webp)/aether-portfolio-enchant-lights-01.webp) type("image/webp"),
    url(var(--asset-png)/aether-portfolio-enchant-lights-01.png) type("image/png")
  );
  background-size: cover;
  background-position: center;
}

Lazy Loading Configuration

/* Native lazy loading */
img[loading="lazy"] {
  opacity: 0;
  transition: opacity 0.3s ease;
}

img[loading="lazy"].loaded,
img[loading="lazy"]:not([src=""]) {
  opacity: 1;
}

/* Intersection Observer enhancement */
<script>
document.querySelectorAll('img[loading="lazy"]').forEach(img => {
  img.addEventListener('load', () => img.classList.add('loaded'));
});
</script>
